import java.util.Scanner;

public class AIChatbot {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.println("AI Chatbot Started (type 'bye' to exit)");

        while (true) {
            System.out.print("You: ");
            String input = sc.nextLine().toLowerCase();

            if (input.equals("bye")) {
                System.out.println("Bot: Goodbye!");
                break;
            } else if (input.contains("hello")) {
                System.out.println("Bot: Hello! How can I help you?");
            } else if (input.contains("name")) {
                System.out.println("Bot: I am a Java AI Chatbot.");
            } else if (input.contains("how are you")) {
                System.out.println("Bot: I'm doing great!");
            } else {
                System.out.println("Bot: Sorry, I don't understand.");
            }
        }
    }
}
